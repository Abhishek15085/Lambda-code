# b) find average price by calarity
# c) Create price category : <200 --- low, 2000 - 5000 --- Medium, >5000 ---high 
# d) Select only price and carat and new column price category



diamonds %>%
  count(color)                             # a) count diamonds for each color

diamonds %>%
  group_by(clarity) %>%                    # b) group by clarity
  summarize(avg_price = mean(price))       # average price

diamonds %>%
  mutate(price_category = case_when(       # c) create price category
    price < 2000 ~ "Low",
    price >= 2000 & price <= 5000 ~ "Medium",
    price > 5000 ~ "High"
  ))

  )) %>%
  select(price, carat, price_category) # select only required columns

#9 Create a function which will take numeric vector as an input and 
# then return sum of all even numbers and sum Of all odd numbers 
# separately test for this vector c(11,13,12,15,16,18, 20)


sum_even_odd <- function(vec) {       # function takes numeric vector
  
  even_sum <- sum(vec[vec %% 2 == 0]) # sum of even numbers
  odd_sum  <- sum(vec[vec %% 2 != 0]) # sum of odd numbers
  
  return(list(even_sum = even_sum,    # return both values
              odd_sum = odd_sum))
}

sum_even_odd(c(11,13,12,15,16,18,20)) # test vector


#10 For numbers 1 to 15
# a) print div3 if numbers is divisible by 3
# b) print div5 if numbers if divisible by 5
# c) print div35 if number iS divisible by both 
# d) print number otherwise 


for (i in 1:15) {                    # loop from 1 to 15
  
  if (i %% 3 == 0 & i %% 5 == 0) {
    print("div35")                  # divisible by both 3 and 5
    
  } else if (i %% 3 == 0) {
    print("div3")                   # divisible by 3
    
  } else if (i %% 5 == 0) {
    print("div5")                   # divisible by 5
    
  } else {
    print(i)                        # otherwise print number
  }
}
